*****frontend/app.py
from flask import Flask
import requests

app = Flask(__name__)

@app.route('/')
def hello_world():
    response = requests.get('http://backend:5000')  # Use 'backend' (not hello-service) as it's the service name
    return f"Frontend says: {response.text}"

if __name__ == '__main__':
    app.run(debug=True, host='0.0.0.0', port=5001)

******docker-compose.yml (at project root)
version: '3.8'

services:
  backend:
    build:
      context: ./backend
    ports:
      - "5000:5000"
    networks:
      - app-network

  frontend:
    build:
      context: ./frontend
    ports:
      - "5001:5001"
    networks:
      - app-network
    depends_on:
      - backend

networks:
  app-network:
    driver: bridge
